FullPage 翻頁效果
FullPage 官網

範例程式碼


        <!-- FullPage.js 區塊 -->
        <div id="fullpage">
          <div class="section section1">Page 1: Welcome</div>
          <div class="section section2">Page 2: About Us</div>
          <div class="section section3">Page 3: Services</div>
          <div class="section section4">Page 4: Contact</div>
        </div>
        
        <!-- FullPage.js 初始化程式 -->
        <script>
          $(document).ready(function() {
            new fullpage('#fullpage', {
              autoScrolling: true,
              navigation: true,
              navigationPosition: 'right',
              navigationTooltips: ['Welcome', 'About Us', 'Services', 'Contact'],
              showActiveTooltip: true
            });
          });
        </script>
