ResponsiveVoice 動畫效果

ResponsiveVoice 官網

範例程式碼

            
                <script src="https://code.responsivevoice.org/responsivevoice.js"></script>
                <button onclick="playText()">播放語音</button>

                <script>
                    function playText() {
                        var text = document.getElementById("text-input").value;
                        if (text) {
                            responsiveVoice.speak("您好！這是一個文字轉語音的範例。希望您會喜歡！", "Chinese Female");
                        } else {
                            alert("請先輸入文字！");
                        }
                    }
                </script>
